./deploy.sh
# Automated steps:
#   1. Builds UI (vite build)

#   3. Restarts tconsiam-api.service
#   4. Runs health check

Nginx Configuration

# /etc/nginx/sites-enabled/tconsiam
server {
    listen 8890;
    server_name _;

    root /opt/tconsiam-dashboard;
    index index.html;

    # SPA fallback — all routes serve index.html
    location / {
        try_files $uri $uri/ /index.html;
    }

    # API reverse proxy
    location /api/ {
        proxy_pass http://127.0.0.1:3457;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }

    # Cache static assets
    location /assets/ {
        expires 30d;
        add_header Cache-Control "public, immutable";
    }
}

Systemd Service

# /etc/systemd/system/tconsiam-api.service
[Unit]
Description=TCONSIAM API Server
After=network.target

[Service]
Type=simple
User=root
WorkingDirectory=/root/projects/tcon/api
ExecStart=/root/.bun/bin/bun run src/index.ts
Restart=on-failure
RestartSec=5
Environment=NODE_ENV=production

[Install]
WantedBy=multi-user.target

Rollback Procedure

Rule: Always backup before deploy. Nothing is Deleted.

Rollback UI

# If UI deploy breaks something:
# Git has the previous build, or use backup
cd /root/projects/tcon/ui
git stash   # Save current changes
git checkout HEAD~1 -- dist/
cp -r dist/* /opt/tconsiam-dashboard/

Rollback Database

systemctl stop tconsiam-api
cp /tmp/tconsiam-pre-deploy.db ~/.config/tconsiam/db/tconsiam.db
systemctl start tconsiam-api

Rollback API

cd /root/projects/tcon/api
git log --oneline -5              # Find last good commit
git checkout <commit-hash> -- src/
systemctl restart tconsiam-api

Troubleshooting

API won't start

  • Check logs: journalctl -u tconsiam-api -n 50
  • Verify Bun installed: bun --version
  • Check port conflict: lsof -i :3457
  • Verify DB file exists: ls ~/.config/tconsiam/db/

Nginx errors

  • Test config: nginx -t
  • Check logs: tail -20 /var/log/nginx/error.log
  • Verify static files: ls /opt/tconsiam-dashboard/index.html

SPA shows blank page

  • Check browser console for JS errors
  • Verify JS/CSS asset files exist in /opt/tconsiam-dashboard/assets/
  • Clear browser cache or hard refresh
  • Check URL uses hash routing: /#/dashboard

502 Bad Gateway

  • API is down — restart: systemctl restart tconsiam-api
  • Port mismatch — verify nginx proxies to correct port
